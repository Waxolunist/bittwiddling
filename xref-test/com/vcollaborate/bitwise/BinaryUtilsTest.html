20  package com.vcollaborate.bitwise;
21  
22  import static org.junit.Assert.assertArrayEquals;
23  import static org.junit.Assert.assertEquals;
24  
25  import org.junit.Test;
26  
27  import java.util.BitSet;
28  
29  public class BinaryUtilsTest {
30  
31    @Test
32    public void testToBitSet() {
33      BitSet bs = new BitSet();
34      bs.set(0, 4);
35      assertEquals(bs, BinaryUtils.toBitSet(4, 15));
36      bs.set(5);
37      assertEquals(bs, BinaryUtils.toBitSet(4, 47));
38    }
39  
40    @Test
41    public void testFromBitSet() {
42      BitSet bs = new BitSet();
43      bs.set(0, 4);
44      assertEquals(15, BinaryUtils.fromBitSet(bs));
45      bs.set(5);
46      assertEquals(47, BinaryUtils.fromBitSet(bs));
47    }
48  
49    @Test
50    public void testNextPermutation() {
51      assertEquals(0b00010101, BinaryUtils.nextPermutation(0b00010011));
52      assertEquals(0b00010110, BinaryUtils.nextPermutation(0b00010101));
53      assertEquals(0b00011001, BinaryUtils.nextPermutation(0b00010110));
54      assertEquals(0b00011010, BinaryUtils.nextPermutation(0b00011001));
55      assertEquals(0b00011100, BinaryUtils.nextPermutation(0b00011010));
56      assertEquals(0b00100011, BinaryUtils.nextPermutation(0b00011100));
57    }
58  
59    @Test
60    public void testGetAllPermutations() {
61      assertEquals(3, BinaryUtils.getAllPermutations(1, 3).length);
62      assertEquals(3, BinaryUtils.getAllPermutations(2, 3).length);
63    }
64  
65    @Test
66    public void testGetHammingDistance() {
67      assertEquals(2, BinaryUtils.getHammingDistance(0b1011101, 0b1001001));
68    }
69  
70    @Test
71    public void testAllOnes() {
72      assertEquals(7, BinaryUtils.allOnes(3));
73    }
74  
75    @Test
76    public void testIsBitSet() {
77      assertEquals(true, BinaryUtils.isBitSet(7, 0));
78      assertEquals(true, BinaryUtils.isBitSet(7, 1));
79      assertEquals(true, BinaryUtils.isBitSet(7, 2));
80      assertEquals(false, BinaryUtils.isBitSet(7, 3));
81    }
82  
83    @Test
84    public void testGetBitsSet() {
85      assertArrayEquals(new int[]{0,1,2}, BinaryUtils.getBitsSet(7));
86      assertArrayEquals(new int[]{3}, BinaryUtils.getBitsSet(8));
87    }
88  
89  }
